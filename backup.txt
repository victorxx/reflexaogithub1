  <script>
    // Lista de vídeos com seus IDs do YouTube e títulos personalizados
    const videos = [
      {id:"0D73FQNkaQ4",titulo:"O NOVO CURSO DE THIAGO FINCH PARA IDIOTAS"},
      {id:"_CoJOENCguc",titulo:"OUTRO QUE FICOU PARA TRÁS NO PASSADO"},
      {id:"Howb7WI1KCk",titulo:"Rica games começou um romance com uma inscrita"},
      {id:"tJ4Og6kjPe8",titulo:"VAMOS RIR! ZOEIRA SEM FIM PALMEIRAS CONTINUA SEM MUNDIAL! KKKK"},
      {id:"Ttp06RN7jTc",titulo:"AS PESSOAS MENTEM PARA VOCÊ!! TODOS OS DIAS!"}
      
    ];

    // Retorna um vídeo aleatório da lista
    function getRandomVideo() {
      return videos[Math.floor(Math.random() * videos.length)];
    }

    // Troca o vídeo exibido e atualiza o título
    function trocarVideo() {
      const propaganda = document.getElementById('propaganda-entre-video');
      propaganda.style.display = "block"; // Exibe a propaganda entre os vídeos

      // Simula um tempo de espera para a propaganda (2 segundos)
      setTimeout(() => {
        const video = getRandomVideo();
        const container = document.getElementById('video-container');
        const tituloEl = document.getElementById('video-title');

        // Esconde a propaganda e carrega o vídeo
        propaganda.style.display = "none";

        container.innerHTML = `<iframe 
          src="https://www.youtube.com/embed/${video.id}" 
          allowfullscreen 
          loading="lazy" 
          title="${video.titulo}"
        ></iframe>`;

        tituloEl.textContent = video.titulo;
      }, 2000); // Exibe a propaganda por 2 segundos antes de carregar o vídeo
    }

    // Gera uma string aleatória com letras, números e caracteres especiais
    function gerarStringUnica(tamanho = 500) {
      const chars = 'ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789!@#$%^&*()_+-=[]{}|;:,.<>?';
      let resultado = '';
      for(let i = 0; i < tamanho; i++) {
        resultado += chars.charAt(Math.floor(Math.random() * chars.length));
      }
      return resultado;
    }

    // Executa ao carregar a página
    document.addEventListener('DOMContentLoaded', () => {
      trocarVideo(); // Carrega o vídeo inicial
      document.getElementById('unique-string').textContent = gerarStringUnica(); // Gera a string aleatória no final
    });
  </script>
